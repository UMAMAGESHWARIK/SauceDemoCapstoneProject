pipeline {
    agent any

    tools {
        jdk 'JDK21'
        maven 'Maven3'
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'master',
                    url: 'https://github.com/UMAMAGESHWARIK/SauceDemoCapstoneProject.git'
            }
        }

        stage('Build & Test') {
            steps {
                bat 'mvn clean test -Dheadless=true'
            }
            post {
                always {
                    archiveArtifacts artifacts: 'test-output/**', fingerprint: true
                    publishHTML([
                        allowMissing: false,
                        alwaysLinkToLastBuild: true,
                        keepAll: true,
                        reportDir: 'test-output',
                        reportFiles: 'ExtentReport.html',
                        reportName: 'Extent Report'
                    ])
                }
            }
        }
    }
}
