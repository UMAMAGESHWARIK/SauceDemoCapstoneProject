pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/UMAMAGESHWARIK/SauceDemoCapstoneProject.git'
            }
        }

        stage('Push Changes') {
            steps {
                echo 'Pushing local changes to GitHub...'
                withCredentials([usernamePassword(credentialsId: 'CapstoneProject', usernameVariable: 'GIT_USER', passwordVariable: 'GIT_TOKEN')]) {
                    bat '''
                        git config --global user.email "umssmageshwari@gmail.com"
                        git config --global user.name "Admin"

                        git add .
                        git commit -m "Automated commit from Jenkins" || echo No changes to commit

                        git push https://%GIT_USER%:%GIT_TOKEN%@github.com/UMAMAGESHWARIK/SauceDemoCapstoneProject.git HEAD:main
                    '''
                }
            }
        }

        stage('Build & Test') {
            steps {
                bat 'mvn clean test -Dsurefire.suiteXmlFiles=testng.xml -Dheadless=true'
            }
        }
    }

    post {
        always {
            archiveArtifacts artifacts: 'test-output/**', fingerprint: true
            publishHTML([
                allowMissing: false,
                alwaysLinkToLastBuild: true,
                keepAll: true,
                reportDir: 'test-output',
                reportFiles: 'ExtentReport.html',
                reportName: 'Extent Report'
            ])
        }
    }
}
